endif()

function(lyon_external name)
    pd_add_external(NAME ${name}
        FILES ${name}.c
        INTERNAL True
        LIBRARY lyonpotpourri
        LINK lyon_static
        HELP_FILES ${name}-help.pd
        )
endfunction()

function(lyon_external_no_help name)
    pd_add_external(NAME ${name}
        FILES ${name}.c
        INTERNAL True
        LIBRARY lyonpotpourri
        LINK lyon_static
        )
endfunction()
