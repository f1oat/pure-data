macro(ceammc_dyn_external name)
    list(APPEND DYN_SOURCES "dyn_${name}.cpp")
    ceammc_faust_gen(dyn ${name})
endmacro()

ceammc_dyn_external(gate2)
ceammc_dyn_external(limit)
