189  static void* createObject(t_symbol*, int argc, t_atom* argv)
190  {
191  ObjectProxy* x = 0;
192  try {
193  x = reinterpret_cast<ObjectProxy*>(pd_new(class_));
194 
195  PdArgs args(AtomList(argc, argv), class_name_, &x->pd_obj);
196  args.noDefaultInlet = flags_ & OBJECT_FACTORY_NO_DEFAULT_INLET;
197  args.mainSignalInlet = flags_ & OBJECT_FACTORY_MAIN_SIGNAL_INLET;
198 
199  x->impl = new T(args);
200  x->impl->parseProperties();
201  } catch (std::exception& e) {
202  x->impl = 0;
203  pd_free(&x->pd_obj.te_g.g_pd);
204  x = 0;
205 
206  char buf[200];
207  snprintf(buf, 199, "%s", e.what());
208  pd_error(0, "[ceammc] can't create object [%s]: %s", class_name_->s_name, buf);
209  } catch (...) {
210  x->impl = 0;
211  pd_free(&x->pd_obj.te_g.g_pd);
212  x = 0;
213 
214  pd_error(0, "[ceammc] can't create object [%s]", class_name_->s_name);
215  }
216 
217  return x;
218  }

250  static void processDataFn(ObjectProxy* x, t_symbol*, int argc, t_atom* argv)
251  {
252  AtomList l(argc, argv);
253  if (l.size() == 1 && l[0].isData()) {
254  DataPtr ptr(l[0]);
255  if (ptr.isValid()) {
256  x->impl->onData(ptr);
257  } else {
258  DataDesc desc = l[0].getData();
259  LIB_ERR << "can't get data with type=" << desc.type << " and id=" << desc.id;
260  }
261  } else {
262  x->impl->onList(l);
263  }
264  }

266  template <class DataT>
267  static void processDataTypedFn(ObjectProxy* x, t_symbol*, int argc, t_atom* argv)
268  {
269  AtomList l(argc, argv);
270  if (l.size() == 1 && l[0].isData()) {
271  DataTPtr<DataT> ptr(l[0]);
272  if (ptr.isValid()) {
273  x->impl->onDataT(*ptr.data());
274  } else {
275  DataDesc d = l[0].getData();
276  LIB_ERR << "can't get data with type=" << d.type << " and id=" << d.id;
277  }
278  } else {
279  x->impl->onList(l);
280  }

293  static void defaultListMethod(ObjectProxy* x, t_symbol* sel, int argc, t_atom* argv)
294  {
295  typename MethodListMap::iterator it = methods_.find(sel);
296  if (it == methods_.end()) {
297  pd_error(x, "unknown method: %s", sel->s_name);
298  return;
299  }
300 
301  (x->impl->*(it->second))(sel, AtomList(argc, argv));
302  }
