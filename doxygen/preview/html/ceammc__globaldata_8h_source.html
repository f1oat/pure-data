28 class NamedDataDict {
29 public:
30  typedef std::pair<size_t, T*> Value;
31  typedef std::map<std::string, Value> Map;
32  typedef typename Map::iterator iterator;
33  typedef typename Map::const_iterator const_iterator;
34 
35 private:
36  Map map_;
37  inline void ref_remove(iterator it) { map_.erase(it); }
38  static inline bool ref_is_zero(iterator it) { return it->second.first == 0; }
39  static inline void ref_free(iterator it) { delete it->second.second; }
40  static inline void ref_inc(iterator it) { ++(it->second.first); }
41  static inline void ref_dec(iterator it) { --(it->second.first); }
42  static T* ref_val(iterator it) { return it->second.second; }
43 
44 public:
45  NamedDataDict() {}

114  {
115  iterator it = map_.find(name);

128  bool release(const std::string& name)
129  {
130  iterator it = map_.find(name);
131  if (it == map_.end())
132  return false;
133 
134  // not acquired value found:
135  if (ref_is_zero(it)) {
136  std::cerr << "[ERROR | NamedDataDict] not acquired element: " << it->first << "\n";
137  ref_free(it);
138  ref_remove(it);
139  return true;
140  }
141 
142  // decrement
143  ref_dec(it);
144 
145  // free is zero
146  if (ref_is_zero(it)) {
147  ref_free(it);
148  ref_remove(it);
149  }
150 
151  return true;
152  }
