109 template <typename T>
110 class EnumProperty : public Property {
111 public:
112  typedef std::vector<T> ValueList;
113 
114 public:
115  EnumProperty(const std::string& name, T def, bool readonly = false)
116  : Property(name, readonly)
117  , idx_(0)
118  , def_(def)
119  {
120  allowed_.push_back(def);
121  }
122 
123  bool set(const AtomList& lst)
124  {
125  if (!readonlyCheck())
126  return false;
127 
128  if (!emptyValueCheck(lst))
129  return false;
130 
131  T v = atomlistToValue<T>(lst, def_);
132  long idx = enumIndex(v);
133  if (idx < 0) {
134  LIB_ERR << "invalid property value: " << v << ". valid values are: " << allowed_;
135  return false;
136  }
137 
138  idx_ = idx;
139  return true;
140  }
141 
142  AtomList get() const
143  {
144  return listFrom(value());
145  }
146 
147  size_t numEnums() const { return allowed_.size(); }
148 
149  T value() const
150  {
151  if (idx_ < 0)
152  return def_;
153  return allowed_[idx_];
154  }
155 
156  bool setValue(T v)
157  {
158  long idx = enumIndex(v);
159  if (idx < 0)
160  return false;
161 
162  idx_ = idx;
163  return true;
164  }
165 
166  void appendEnum(T v)
167  {
168  if (enumIndex(v) < 0)
169  allowed_.push_back(v);
170  }
171 
172  long enumIndex(T v) const
173  {
174  typename ValueList::const_iterator it;
175  for (it = allowed_.begin(); it != allowed_.end(); ++it) {
176  if (*it == v)
177  return std::distance(allowed_.begin(), it);
178  }
179  return -1;
180  }
181 
182 private:
183  ValueList allowed_;
184  T def_;
185  int idx_;
186 };
